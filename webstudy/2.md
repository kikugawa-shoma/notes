# フォームに入力させた2数の積を表示する
## ディレクトリ構成
```
test
    │  index.html
    │
    └─ cgi-bin
            test.py
```

## ファイル
index.html
```html
<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <title>テストページ</title>
    </head>
    <body>
        <h1>こんにちは</h1>
        2つの数a,bを入力してください。
        <form method="POST" action="cgi-bin/test.py">
            <label>a : </label>
            <textarea name="a"></textarea>
            <label>b : </label>
            <textarea name="b"></textarea>
            <button type="submit">決定</button>
        </form>
    </body>
</html>
```


cgi-bin/test.py
```python
#! C:\Users\ktmks\anaconda3\envs\webapp\python
# -*- coding: utf-8 -*-

import cgi
import cgitb
import sys

cgitb.enable()

form = cgi.FieldStorage()

print("<meta charset=`utf-8`>")
print("")

if "a" not in form or "b" not in form:
    print("<h1>Error!</h1>")
    print("<br>")
    print("テキストを入力してください！")
    print("<a href='/'><button type='submit'>戻る</button></a>")
    sys.exit()

else:
    print("aとbの積は")
    a = int(form.getvalue("a"))
    b = int(form.getvalue("b"))
    print(a*b)
    print("です。")
```
